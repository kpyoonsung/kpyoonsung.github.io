java 복습
~~~
int number;   //정수형 변수 number선언
number = 20;
System.out.println(number);  //number를 출력
System.out.println(number+10); //number+10 값을 출력


double a = 4.14;   //실수형 변수 a의 값을  선언
System.out.println(a);  //a의 값을 출력
System.out.println((int)a);   //a를 형변환에서 대입

int num1,num2;  //num1과 num2를 선언
num1=4;
num2=14;
System.out.println(num1+num2); //두 변수의 합을 출력
System.out.println(num1*num2);  //두 변수의 곱을 출력
~~~
